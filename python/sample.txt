# Read the first line containing n and m
n, m = map(int, input().split())

# Initialize an empty 2-dimensional array
array = []

# Read the next n lines and append them to the array
for _ in range(n):
    row = list(map(int, input().split()))
    array.append(row)

# Print the resulting 2-dimensional array
for row in array:
    print(row)
